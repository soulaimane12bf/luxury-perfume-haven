# replace-secrets.txt for git-filter-repo
# Format: <string-or-regex>==><replacement>
# Use regex: prefix to redact patterns without embedding exact secret values.

# redact Neon/Postgres connection strings
regex:postgresql://[^\s"']+==>[REDACTED_DB_URL]

# redact Vercel blob tokens
regex:vercel_blob_rw_[A-Za-z0-9_-]+==>[REDACTED_VERCEL_BLOB]

# redact JWT-like tokens that start with eyJ
regex:eyJ[a-zA-Z0-9._-]+==>[REDACTED_JWT]

# redact any PGPASSWORD or POSTGRES_PASSWORD values in env-like lines
regex:PGPASSWORD\s*=\s*[^\s"']+==>[REDACTED_PGPASSWORD]
regex:POSTGRES_PASSWORD\s*=\s*[^\s"']+==>[REDACTED_PGPASSWORD]

# redact any DATABASE_URL-like full URLs
regex:(?:DATABASE_URL|POSTGRES_URL|POSTGRES_PRISMA_URL|POSTGRES_URL_NON_POOLING)\s*=\s*[^\n\r]+==>[REDACTED_DB_URL]
